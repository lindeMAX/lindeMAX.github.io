Python

The knowledge gained from the article can be used, to do something (actually not so) usefull. But you’ll get the idea ;-).

The script below will create four different sine functions. They’ll have different circular frequencies and to one of them white gaussian noise is added. They do then get plotted in time and frequency domain.

import numpy as np
import matplotlib.pyplot as plt
import matplotlib.ticker as tck
from scipy.fft import fft, fftfreq

# Time period of the samples
# This is directly related to the frequencies that can be dissolved!
# Check Shannon-Nyquist sampling theorem for more information!
# As one can see in the comment to the 'Signal length',
# a to high sample rate will actually reduce the frequency resolution.
# -> Sample only as high, as you need to!
# This seems counter intuituve, but so it is. Trust me ;-)
f_max = 60 # in Hz
T = 1/(2 * f_max)

# Signal length [s]
# You'll need a long time interval for a good resolution in freequency domain!
# The spacing of the frequencies (DFT!) is given by delta_f = 1/ml = 1/(N*T).
# You can actually see that by reducing the singal length or increasing the sampling frequeny.
sl = 100 # Seconds

# Total number of samples
N = int(sl / T)

# Time axis values
t = np.linspace(0, sl, N)

# Circular frequencies
omega0 = 2 * np.pi * 5
omega1 = 2 * np.pi * 10
omega2 = 2 * np.pi * 50

# Signals in time domain
y0 = np.sin(omega0*t)
y1 = np.sin(omega0*t) + np.sin(omega1*t) + np.sin(omega2*t)
y2 = np.sin(omega0*t) + 2*np.sin(omega1*t) + 4*np.sin(omega2*t)

# GWN:
# - uniform power across the frequency band
# - normaly distributed in time domain
noise = np.random.normal(0,5,N)
y3 = np.sin(omega0*t) + 2*np.sin(omega1*t) + 4*np.sin(omega2*t) + noise

# Frequency spectra (FFT / DFT)
yf0 = fft(y0)
yf1 = fft(y1)
yf2 = fft(y2)
yf3 = fft(y3)

# Frequency axis values
xf = fftfreq(y0.size, T)

fig, ax = plt.subplots(4, 2, figsize=(20,15))

ax[0][0].set_xlabel('Time $[s]$ ')
ax[0][0].xaxis.set_major_formatter(tck.FormatStrFormatter('%g s'))
ax[0][0].yaxis.set_major_locator(tck.MultipleLocator(base=0.5))
# Plotting only the first second (visually nicer)
ax[0][0].plot(t[:int(1/T)], y0[:int(1/T)])
ax[0][0].grid()

ax[0][1].set_xlabel('Frequency $[Hz]$')
# Plotting only the upper half of the freq-domain
# Be careful with the y-axis scaling!
ax[0][1].plot(xf[:N//2], 2.0/N * np.abs(yf0[:N//2]))
ax[0][1].grid()

ax[1][0].set_xlabel('Time $[s]$ ')
ax[1][0].xaxis.set_major_formatter(tck.FormatStrFormatter('%g s'))
ax[1][0].yaxis.set_major_locator(tck.MultipleLocator(base=0.5))
ax[1][0].plot(t[:int(1/T)], y1[:int(1/T)])
ax[1][0].grid()

ax[1][1].set_xlabel('Frequency $[Hz]$')
ax[1][1].plot(xf[:N//2], 2.0/N * np.abs(yf1[0:N//2]))
ax[1][1].grid()

ax[2][0].set_xlabel('Time $[s]$ ')
ax[2][0].xaxis.set_major_formatter(tck.FormatStrFormatter('%g s'))
ax[2][0].yaxis.set_major_locator(tck.MultipleLocator(base=1))
ax[2][0].plot(t[:int(1/T)], y2[:int(1/T)])
ax[2][0].grid()

ax[2][1].set_xlabel('Frequency $[Hz]$')
ax[2][1].plot(xf[:N//2], 2.0/N * np.abs(yf2[0:N//2]))
ax[2][1].grid()

ax[3][0].set_xlabel('Time $[s]$ ')
ax[3][0].xaxis.set_major_formatter(tck.FormatStrFormatter('%g s'))
ax[3][0].yaxis.set_major_locator(tck.MultipleLocator(base=1))
ax[3][0].plot(t[:int(1/T)], y3[:int(1/T)])
ax[3][0].grid()

ax[3][1].set_xlabel('Frequency $[Hz]$')
ax[3][1].plot(xf[:N//2], 2.0/N * np.abs(yf3[0:N//2]))
ax[3][1].grid()

fig.tight_layout()
